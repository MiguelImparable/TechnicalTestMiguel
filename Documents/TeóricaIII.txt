// 6.Dado:
public class Starter extends Thread {
    private int X = 2;

    public static void main(String[] args) throws Exception {
        new Starter().makeltSo();
    }

    public Starter() {
        X = 5;
        start();
    }

    public void makeltSo() throws Exception {
        join();
        X = X - 1;
        System.out.println(X);
    }

    public void run() {
        X *= 2;
    }
}
// ¿Cuál es el resultado?
// D. 9
